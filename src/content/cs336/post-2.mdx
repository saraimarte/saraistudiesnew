<Code lang = "sql" fileName = "cs336.sql" code = {
`-- 1. Create a new database for your bank
CREATE DATABASE bank_db;

-- 2. Switch to the newly created database to work inside it
USE bank_db;

-- 3. Create the 'users' table
CREATE TABLE users (
  name VARCHAR(100) primary key,
  age INT,
  amount DECIMAL(10, 2)
);

-- 4. Insert some sample values into the 'users' table
INSERT INTO users (name, age, amount) VALUES 
('Alice Smith', 30, 1500.75), 
('Bob Johnson', 25, 800.00),
('Charlie Brown', 42, 10200.50);

-- 5. Selecting all data from the table
SELECT * FROM users;`}/>
