create database university;
use university;

create table students(
sid int,
sname varchar(30),
major varchar(30),
level varchar(10),
age int,
primary key (sid));

create table faculty(
fid int,
fname varchar(30),
dept varchar(4),
primary key (fid));
create table courses(
cname varchar(30),
meets_at time,
room varchar(10),
fid int,
primary key (cname),
foreign key (fid) references faculty(fid));

create table enrolledin(
sid int,
cname varchar(30),
primary key (sid,cname),
foreign key (sid) references students(sid),
foreign key (cname) references courses(cname));
