<Problem n = "6">

</Problem>

<Problem n = "7">

<Problem n = "8">
<Latex formula = "\int e^{4x} \cos(x) \, dx"/>
</Problem>
<Solution>
<Latex formula = "\frac{1}{17} e^{4x} \sin(x) + \frac{4}{17} e^{4x} \cos(x) + C"/>
</Solution>


<Problem n = "9">
<Latex formula = "\int_{1}^{e} \frac{\ln^2(x)}{x^4} \, dx"/>
</Problem>
<Solution>
<Latex formula = "\frac{2e^{3} - 17}{27e^{3}}"/>
</Solution>

<Problem n = "10">

</Problem>

</Solution>
